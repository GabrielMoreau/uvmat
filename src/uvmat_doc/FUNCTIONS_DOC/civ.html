3763 %---------------------------------------
3764 function [nbslice, flag_mask]=get_mask(filebase,handles)
3765 %detect mask files, images with appropriate file base
3766 %[filebase '_' xx 'mask'], xx=nbslice
3767 %flag_mask=1 indicates detection
3768 
3769 flag_mask=0;%default
3770 nbslice=1;
3771 
3772 % subdir=get(handles.subdir_civ1,'String');
3773 [Path,Name]=fileparts(filebase);
3774 currentdir=pwd;
3775 cd(Path);%move in the dir of the root name filebase
3776 maskfiles=dir([Name '_*mask_*.png']);%look for mask files
3777 cd(currentdir);%come back to the current working directory
3778 if isempty(maskfiles)
3779     msgbox_uvmat('ERROR','no mask available, to create it use Tools/Make mask in the upper menu bar of uvmat')
3780 else
3781     flag_mask=1;
3782     maskname=maskfiles(1).name;% take the first mask file in the list
3783     [Path2,Name,ext]=fileparts(maskname);
3784     Namedouble=double(Name);
3785     val=(48>Namedouble)|(Namedouble>57);% select the non-numerical characters
3786     ind_mask=findstr('mask',Name);
3787     i=ind_mask-1;
3788     while val(i)==0 & i>0
3789        i=i-1;
3790     end
3791     nbslice=str2num(Name(i+1:ind_mask-1));
3792     if ~isequal(nbslice,[]) & Name(i)=='_'
3793           flag_mask=1;
3794     else
3795           msgbox_uvmat('ERROR',['bad mask file ' Name ext ' found in ' Path2])
3796           return
3797           nbslice=1;
3798     end
3799 end
